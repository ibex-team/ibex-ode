variables
  x[2] in (0; 0);

function f(x[2])
  return(x(2); 5*max(-1, min(1, -2*x(1) - x(2))) + x(1));
end

function v(u[2])
  return(u(1)*(1.2777777777777777*u(1) + 0.05555555555555555*u(2)) + (0.05555555555555555*u(1) + 0.1111111111111111*u(2))*u(2) + (u(1) + u(2))^4/15);
end

function vminor(u[2])
  return(u(1)*(1.2777777777777777*u(1) + 0.05555555555555555*u(2)) + (0.05555555555555555*u(1) + 0.1111111111111111*u(2))*u(2));
end
